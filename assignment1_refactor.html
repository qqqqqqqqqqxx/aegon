# Helper functions 
def pv_of_bond_changing_rate(cp:float, fv:float, maturity:int, spread=0, inter_func: Callable=None):
    """Refactor of above function: Basic function for pv with changing rate"""
    pv = 0
    coupon = cp / 100 * fv
    
    for t in range(1, maturity + 1):
        discount_factor = (1 + inter_func(t)/100 + spread)**(-t)
        pv += coupon * discount_factor

        # Add nominal at maturity
        if t == maturity:
            pv += fv * discount_factor
    return pv

def get_price(x, inter_func: Callable=None):
    return pv_of_bond_changing_rate(cp=x['Coupon'], fv=100, maturity=x['Maturity'], spread=0, inter_func=inter_func)

def z_spread(cp, fv, maturity, price):
    """Solving spread"""
    equ = lambda x: pv_of_bond_changing_rate(cp, fv, maturity, spread=x) - price
    # Initial guess for the spread is 0
    spread = fsolve(equ, 0)[0]
    return spread

def apply_to_all(x):
    """apply to all df"""
    # try to solve the implied Z
    return z_spread(cp=x['Coupon'], fv=100, maturity=x['Maturity'], price=x['Price'])
